```js
// src/lib/box2d-init.js
//
// Loads lib/box2d.js in a way that preserves its global-assignment behavior.
//
// Why this works:
//   box2d.js ends with: if(typeof exports !== "undefined") { exports.b2* = ... }
//   We import its URL (not its content) via the ?url Vite suffix, then inject
//   it as a classic <script> tag. Classic scripts have no module scope and no
//   esbuild CJS transform — the b2* constructors land on window/globalThis.
//
// This file itself is NOT transformed by esbuild CJS detection because it
// contains no top-level 'exports.' assignments — only a URL import and DOM API.

import box2dUrl from "/lib/box2d.js?url";

export async function loadBox2d() {
  // Execute as classic script (not module) so 'exports' stays undefined
  // and all b2* constructors land on globalThis as intended.
  await new Promise((resolve, reject) => {
    const script = document.createElement("script");
    script.src = box2dUrl;
    script.onload = resolve;
    script.onerror = reject;
    document.head.appendChild(script);
  });
}
```

**Create `src/main.js`** with this exact content:

```js
// src/main.js
//
// Vite module entry point. Sequences dependency loading before the app IIFE runs.
//
// Load order (critical):
//   1. seedrandom.js — patches Math.seedrandom (static import = hoisted, runs first)
//   2. loadBox2d()   — injects box2d.js as a classic script, populates window.b2*
//   3. app.js        — dynamic import so it runs AFTER box2d globals are on window
//
// Why dynamic import for app.js:
//   Static imports are hoisted and resolved before any await expressions run.
//   Using await import('./app.js') ensures app.js executes strictly after
//   loadBox2d() resolves — guaranteeing b2Vec2, b2World, etc. exist when the
//   IIFE inside app.js accesses them as free identifiers.

import "../lib/seedrandom.js"; // side-effect: patches Math.seedrandom
import { loadBox2d } from "./lib/box2d-init.js";

await loadBox2d(); // populates window.b2World, window.b2Vec2, etc.
await import("./app.js"); // IIFE executes; b2* already on window
```
